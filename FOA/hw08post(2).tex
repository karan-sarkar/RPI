\begin{enumerate}

\item Use the definition of {\bf limit of $f$ at $c$} to prove that 
$$ \lim_{x\to 3} \frac{x^2+4x-21}{x-3} = 10.$$

\item Use the definition of {\bf continuity of $f$ at $c$} to prove that 
$f(x) = x^2 - 5x+1$ is continuous at $4$ .

\end{enumerate}
